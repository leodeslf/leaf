GitHub Pages

# Install
npm i -D gh-pages
# Use
gh-pages -d <dir> -m <commit-message>
# Scripts
npm run predeploy
npm run deploy
// package.json
{
  "scripts": {
    "predeploy": "npm run build",
    "deploy": "gh-pages -d build"
  },
  "homepage": "https://<user-name>.github.io/<repo-name>"
}
