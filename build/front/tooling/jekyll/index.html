Jekyll

# Install
gem install jekyll bundler
# Use
jekyll new <project-name>
# To fix 3.0.0+ Ruby
bundle add webrick
# Bundle
bundle exec jekyll build
bundle exec jekyll serve
# `bundle exec` is optional after the first use.
