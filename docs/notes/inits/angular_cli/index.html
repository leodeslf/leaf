Angular CLI

npm i -g @angular/cli
ng new <project-name>
ng generate component <component-name>
ng add <dependency-name>
npm start
npm run build
// package.json
{
  "scripts": {
    "start": "ng serve --port <number> --open",
    "build": "ng build --prod",
  }
}
