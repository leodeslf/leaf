NPM Packages

npmrc <usr-name>
npm init --scope=@<usr-name>
npm publish --access public
npm version [major | minor | patch]
npm publish
npm unpublish <pkg-name> -f
// package.json
{
  "name": "@<usr-name>/<pkg-name>",
  "publishConfig": {
    "registry": "http://registry.npmjs.org",
    "access": "public"
  }
}
